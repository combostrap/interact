import {getPages} from "../.interact/App";

# Index

<p>Number of Pages: {Object.keys(getPages()).length-1}</p>
<ul>
    {Object.entries(getPages()).map(([key, value]) => {
        if(key==="/index"){
            return;
        }
        return (
            <li key={key}>
                <a href={key} style={{textTransform: 'capitalize'}}>
                    {(value).title ?? key.slice(1)}
                </a>
            </li>
        )
    })}
</ul>
